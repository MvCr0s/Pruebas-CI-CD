name: Pipeline CI/CD Completo

on:
  push:
    branches: ["main"]

jobs:
  # TRABAJO 1: INTEGRACIÓN CONTINUA (CI)
  test:
    runs-on: ubuntu-latest
    steps:
      - name: Descargar código
        uses: actions/checkout@v3

      - name: Configurar Python
        uses: actions/setup-python@v4
        with:
          python-version: "3.9"

      - name: Instalar dependencias
        run: |
          python -m pip install --upgrade pip
          pip install -r app/requirements.txt

      - name: Ejecutar Tests (Pytest)
        run: |
          # Esto buscará archivos que empiecen por test_ y verificará que todo funciona
          python -m pytest

  # TRABAJO 2: DESPLIEGUE CONTINUO (CD)
  build-and-push:
    needs: test # IMPORTANTE: Esto solo arranca si 'test' sale VERDE
    runs-on: ubuntu-latest
    steps:
      - name: Descargar código
        uses: actions/checkout@v3

      - name: Login en Docker Hub
        uses: docker/login-action@v2
        with:
          username: ${{ secrets.DOCKER_USERNAME }}
          password: ${{ secrets.DOCKER_PASSWORD }}

      - name: Construir y Subir imagen Docker
        uses: docker/build-push-action@v4
        with:
          context: .
          push: true
          # Aquí se guardará como usuario/repo:etiqueta
          tags: ${{ secrets.DOCKER_USERNAME }}/gamemood-tfg:latest
